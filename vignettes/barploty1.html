Теперь модификация таблиц:

bolshaja$participants[bolshaja$participants=='zveri']<-'zhivotnye'
bolshaja$participants[bolshaja$participants=='prochee']<-'neopoznannye'
bolshaja_moj<-subset(bolshaja, grepl("moj",pattern))
bolshaja_bez_moj<-subset(bolshaja,!grepl("moj",pattern))
tablica_moj<-table(bolshaja_moj$participants, bolshaja_moj$otdelno)
tablica_bez_moj<-table(bolshaja_bez_moj$participants, bolshaja_bez_moj$otdelno)
bolshaja$otdelno<-apply(bolshaja,1,function(r)gsub("\\[adj\\]",r[["adj"]],r[["pattern"]]))
tablica_otdelno<-table(bolshaja$participants, bolshaja$otdelno)

И теперь графики. Обратите внимание на параметр vertical=T:

PrintProfilePlot(tablica_otdelno, c("lubimyj moj", "rodnoj"), vertical=T)

Вот втрой рисунок, тот который был пустым:

PrintProfilePlot(tablica_moj,c("lubimyj moj", "milyj moj"), vertical=T)
